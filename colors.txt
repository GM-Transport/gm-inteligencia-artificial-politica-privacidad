import 'package:flutter/material.dart';
import 'package:google_fonts/google_fonts.dart';

class AppTextStyles {
  // Helper para crear TextStyle con fallback
  static TextStyle _createTextStyle({
    required double fontSize,
    required FontWeight fontWeight,
    double letterSpacing = 0,
    double height = 1.0,
  }) {
    try {
      return GoogleFonts.dmSans(
        fontSize: fontSize,
        fontWeight: fontWeight,
        letterSpacing: letterSpacing,
        height: height,
      );
    } catch (e) {
      // Fallback a fuente del sistema si Google Fonts falla
      return TextStyle(
        fontSize: fontSize,
        fontWeight: fontWeight,
        letterSpacing: letterSpacing,
        height: height,
        fontFamily: 'Segoe UI', // Windows default
      );
    }
  }

  // Display styles
  static TextStyle get displayLarge => _createTextStyle(
    fontSize: 57,
    fontWeight: FontWeight.w400,
    letterSpacing: -0.25,
    height: 1.12,
  );

  static TextStyle get displayMedium =>
      _createTextStyle(fontSize: 45, fontWeight: FontWeight.w400, height: 1.16);

  static TextStyle get displaySmall =>
      _createTextStyle(fontSize: 36, fontWeight: FontWeight.w400, height: 1.22);

  // Headline styles
  static TextStyle get headlineLarge =>
      _createTextStyle(fontSize: 32, fontWeight: FontWeight.w600, height: 1.25);

  static TextStyle get headlineMedium =>
      _createTextStyle(fontSize: 28, fontWeight: FontWeight.w600, height: 1.29);

  static TextStyle get headlineSmall =>
      _createTextStyle(fontSize: 24, fontWeight: FontWeight.w600, height: 1.33);

  // Title styles

      _createTextStyle(fontSize: 22, fontWeight: FontWeight.w500, height: 1.27);

  static TextStyle get titleMedium => _createTextStyle(
    fontSize: 16,
    fontWeight: FontWeight.w500,
    letterSpacing: 0.15,
    height: 1.50,
  );

  static TextStyle get titleSmall => _createTextStyle(
    fontSize: 14,
    fontWeight: FontWeight.w500,
    letterSpacing: 0.1,

  );

  // Label styles

  );

  static TextStyle get bodyLarge => _createTextStyle(

    height: 1.50,
  );

  static TextStyle get bodyMedium => _createTextStyle(
    fontSize: 14,
    fontWeight: FontWeight.w400,
    letterSpacing: 0.25,
    height: 1.43,
  );

  static TextStyle get bodySmall => _createTextStyle(
    fontSize: 12,
    fontWeight: FontWeight.w400,
    letterSpacing: 0.4,
    height: 1.33,
  );

  // Custom styles para IA
  static TextStyle get aiTitle {
    try {
      return GoogleFonts.orbitron(
        fontSize: 24,
        fontWeight: FontWeight.w700,
        letterSpacing: 1.0,
      );
    } catch (e) {
      return const TextStyle(
        fontSize: 24,
        fontWeight: FontWeight.w700,
        letterSpacing: 1.0,
        fontFamily: 'Consolas', // Windows monospace fallback
      );
    }
  }

  static TextStyle get aiSubtitle {
    try {
      return GoogleFonts.orbitron(
